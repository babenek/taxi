select
  count( ORDERS.TIME_START) cnt
from 
  SMENA, ORDERS
where
  (ORDERS.KEY_ORD_TYPE = :ORD) and
  (SMENA.SMN_DATE>=:D1) and (SMENA.SMN_DATE<=:D2)
  and  SMENA.UNIQS=ORDERS.KEY_SMN_START
  and ORDERS.ORD_STATE in (1,3)
--group by  SMENA.UNIQS, SMENA.SMN_DATE, SMENA.USR_LIST,        SMENA.SMN_DAY_NIGHT
